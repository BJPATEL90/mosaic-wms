const GAS_URL_KEY = 'gasUrl';
function gasUrl(){ return ls(GAS_URL_KEY) || ''; }

async function call(action, payload, showErr){
  // Method 1: inside Apps Script iframe
  if(typeof google !== 'undefined' && google.script){
    return new Promise((res,rej) => {
      google.script.run
        .withSuccessHandler(r => res(r))
        .withFailureHandler(e => rej(new Error(e.message||'Script error')))
        .doPost_({action, payload});
    }).catch(e => {
      if(showErr !== false) console.warn('GAS error:',e.message);
      return {success:false, message:e.message};
    });
  }
  // Method 2: fetch to deployed URL
  const url = gasUrl();
  if(!url) return {success:false, message:'GAS URL not set — data saved locally only.'};
  try {
    const r = await fetch(url, {method:'POST', headers:{'Content-Type':'text/plain'}, body: JSON.stringify({action, payload})});
    const t = await r.text();
    try { return JSON.parse(t); } catch(e) { return {success:false, message:t}; }
  } catch(e){ return {success:false, message:e.message}; }
}

function updateGRNBreakdown(){
  const qty=Number($('g-qty').value)||0;
  const cp=Number($('g-cp').value)||0;
  const bp=Number($('g-bp').value)||0;
  const el=$('g-breakdown');
  if(!qty){el.style.display='none';return;}
  const {fullBoxes,looseUnits,totalBoxes,binsNeeded}=boxBreakdown(qty,cp,bp);
  el.style.display='block';
  el.innerHTML=`<b>Breakdown:</b> ${fullBoxes} full boxes + ${looseUnits} loose → ${totalBoxes} total boxes → <b>${binsNeeded} bin(s) needed</b>
    ${bp>0?'<br>Bin allocation: '+Array.from({length:binsNeeded},(_, i)=>{const boxes=i<binsNeeded-1?bp:totalBoxes-i*bp;const units=i<binsNeeded-1?boxes*cp:looseUnits>0?(boxes-1)*cp+looseUnits:boxes*cp;return`Bin ${i+1}: ${boxes} boxes (${units} units)`;}).join(' | '):''}`;
}

function boxBreakdown(qty,cp,bp){
  qty=Number(qty)||0;cp=Number(cp)||0;bp=Number(bp)||0;
  if(!qty)return{fullBoxes:0,looseUnits:0,totalBoxes:0,binsNeeded:1};
  const fullBoxes=cp>0?Math.floor(qty/cp):0;
  const looseUnits=cp>0?qty%cp:qty;
  const totalBoxes=fullBoxes+(looseUnits>0?1:0);
  const binsNeeded=bp>0?Math.ceil(totalBoxes/bp):1;
  return{fullBoxes,looseUnits,totalBoxes,binsNeeded};
}

function parseGRNFile(ev){
  const f=ev.target.files[0];if(!f)return;
  const r=new FileReader();
  r.onload=e=>{
    const wb=XLSX.read(new Uint8Array(e.target.result),{type:'array'});
    const rows=XLSX.utils.sheet_to_json(wb.Sheets[wb.SheetNames[0]],{defval:''});
    const h=rows.length?Object.keys(rows[0]).map(k=>k.toLowerCase()):[];
    const fc=(als)=>Object.keys(rows[0]||{}).find(k=>als.includes(k.toLowerCase().trim()))||null;
    const cm={
      sku:fc(['sku code','sku_code','material code','sku']),
      name:fc(['sku name','sku_name','material name','name']),
      brand:fc(['brand']),
      batch:fc(['batch','batch no','batch_no','lot']),
      qty:fc(['qty','quantity','total qty']),
      cp:fc(['case pack','casepack','units per box']),
      bp:fc(['box/pallet','box pallet','boxes per shelf']),
      cls:fc(['fsn class','class','classification'])
    };
    let added=0;
    rows.forEach(r=>{
      const sku=String(r[cm.sku]||'').trim();
      const qty=Number(String(r[cm.qty]||'0').replace(/,/g,''));
      if(!sku||qty<=0)return;
      const skuRow=skuMaster.find(s=>(s['SKU Code']||s.code||'')==sku)||{};
      const cp=Number(cm.cp?r[cm.cp]:0)||Number(skuRow['Case Pack']||skuRow.casePack)||0;
      const bp=Number(cm.bp?r[cm.bp]:0)||Number(skuRow['Box/Pallet']||skuRow.boxPallet)||0;
      const cls=cm.cls?String(r[cm.cls]).trim():skuRow['FSN Class']||skuRow.cls||'B';
      const brand=cm.brand?String(r[cm.brand]).trim():skuRow['Brand']||skuRow.brand||'';
      const bd=boxBreakdown(qty,cp,bp);
      grnLines.push({skuCode:sku,skuName:cm.name?String(r[cm.name]).trim():skuRow['SKU Name']||skuRow.name||'',
        brand,batch:cm.batch?String(r[cm.batch]).trim():'',qty,casePack:cp||null,boxPallet:bp||null,cls:cls||'B',...bd});
      added++;
    });
    $('g-upload-msg').innerText='✓ '+added+' line(s) loaded from '+f.name;
    renderGRNTable();
  };
  r.readAsArrayBuffer(f); ev.target.value='';
}

async function parseDumpFile(ev){
  const f=ev.target.files[0];if(!f)return;
  const msg=$('dump-msg'); msg.innerText='Reading…';
  const reader=new FileReader();
  reader.onload=async e=>{
    const wb=XLSX.read(new Uint8Array(e.target.result),{type:'array'});
    const rows=XLSX.utils.sheet_to_json(wb.Sheets[wb.SheetNames[0]],{defval:''});
    if(!rows.length){msg.innerText='No rows found';return;}
    const heads=Object.keys(rows[0]);
    const cm={}; Object.keys(DUMP_COLS).forEach(k=>cm[k]=findCol(heads,DUMP_COLS[k]));
    if(!cm.skuCode||!cm.qty){msg.innerText='SKU Code and Qty columns required';return;}
    dumpRows=rows.map(r=>({
      skuCode:String(r[cm.skuCode]||'').trim(),
      skuName:cm.skuName?String(r[cm.skuName]||'').trim():'',
      brand:cm.brand?String(r[cm.brand]||'').trim():'',

      bin:cm.bin?String(r[cm.bin]||'').trim():'',
      stockType:cm.stockType?String(r[cm.stockType]||'GOOD').trim().toUpperCase():'GOOD',
      qty:Number(String(r[cm.qty]||'0').replace(/,/g,'')),
      ean:cm.ean?String(r[cm.ean]||'').trim():'',
    })).filter(r=>r.skuCode&&r.qty>0);
    inventory=dumpRows; lss('inv',inventory);

    syncBinsLocal();
    msg.innerText=`✓ ${dumpRows.length} rows loaded from ${f.name}. Bin status reset from dump. Syncing to Google Sheet…`;

function syncBinsLocal(){
  const occ={};
  dumpRows.forEach(r=>{ if(r.bin&&r.qty>0){ if(!occ[r.bin])occ[r.bin]=[]; if(!occ[r.bin].includes(r.skuCode))occ[r.bin].push(r.skuCode); } });
  binMaster.forEach(b=>{
    const id=b['Bin ID']||b.id||b.binId||'';
    const st=String(b['Status']||b.status||'').toUpperCase();
    if(occ[id]){ b['Status']=b.status='OCCUPIED'; b['Occupied SKU']=b.skusInBin=occ[id].slice(0,2).join(', ');}
    else if(st==='OCCUPIED'){ b['Status']=b.status='EMPTY'; b['Occupied SKU']=b.skusInBin=''; }
  });
  lss('bin',binMaster);
}

async function parseBinFile(ev){
  const f=ev.target.files[0];if(!f)return;
  const msg=$('bin-upload-msg'); msg.innerText='Reading…';
  const reader=new FileReader();
  reader.onload=async e=>{
    const wb=XLSX.read(new Uint8Array(e.target.result),{type:'array'});
    const rows=XLSX.utils.sheet_to_json(wb.Sheets[wb.SheetNames[0]],{defval:''});
    if(!rows.length){msg.innerText='No rows found';return;}
    binMaster=rows.filter(r=>r['Bin ID']||r['Bin Code']||r.binId).map(r=>({
      'Bin ID':String(r['Bin ID']||r['Bin Code']||r.binId||'').trim(),
      'Rack':String(r['Rack']||r.rack||'').trim(),
      'Level':String(r['Level']||r.level||'').trim(),
      'Capacity (Boxes)':Number(r['Capacity (Boxes)']||r.capacity||'')||'',
      'Status':String(r['Status']||r.status||'EMPTY').trim().toUpperCase(),
      'Occupied SKU':String(r['Occupied SKU']||r.occupiedSKU||'').trim(),
      'Last Updated':new Date().toISOString()
    }));
    lss('bin',binMaster);
    msg.innerText=`✓ ${binMaster.length} bins loaded from ${f.name}. Syncing…`;
    filterBins();
    const r=await call('saveBinMaster',binMaster);
    msg.innerText=r&&r.success?`✓ ${binMaster.length} bins synced to Google Sheet.`:`✓ Saved locally. ⚠ Sheet sync: ${r&&r.message||''}`;
  };
  reader.readAsArrayBuffer(f); ev.target.value='';
}
function filterBins(){
  const q=$('bin-search').value.toLowerCase();
  const sf=String($('bin-filter').value||'').toUpperCase();
  renderBinTable(binMaster.filter(b=>{
    const id=String(b['Bin ID']||b.id||'');
    const st=String(b['Status']||b.status||'').toUpperCase();
    return (id+' '+String(b['Rack']||b.rack||'')).toLowerCase().includes(q)&&(!sf||st===sf);
  }));
  const chips=$('bin-chips');
  const total=binMaster.length,occ=binMaster.filter(b=>String(b['Status']||b.status||'').toUpperCase()==='OCCUPIED').length,emp=binMaster.filter(b=>String(b['Status']||b.status||'').toUpperCase()==='EMPTY').length,res=binMaster.filter(b=>String(b['Status']||b.status||'').toUpperCase()==='RESERVED').length;
  chips.innerHTML=`<span class="chip chip-t">${total} Total</span><span class="chip chip-ok">${emp} Empty</span><span class="chip chip-err">${occ} Occupied</span><span class="chip" style="background:var(--amber-l);color:var(--amber)">${res} Reserved</span>`;
}
function renderBinTable(rows){
  rows=rows||binMaster;
  const body=$('bin-body');
  if(!rows.length){body.innerHTML='<tr class="er"><td colspan="8">No bins</td></tr>';return;}
  body.innerHTML=rows.map((b,i)=>{
    const id=b['Bin ID']||b.id||'';
    const st=String(b['Status']||b.status||'EMPTY').toUpperCase();
    return`<tr>
      <td><span class="bin-b">${id}</span></td>
      <td>${b['Rack']||b.rack||'—'}</td><td>${b['Level']||b.level||'—'}</td>
      <td style="text-align:center">${b['Capacity (Boxes)']||b.capacity||'—'}</td>
      <td><span class="badge ${st==='OCCUPIED'?'b-occ':st==='RESERVED'?'b-res':'b-empty'}">${st}</span></td>
      <td style="font-size:11px;color:var(--slate4)">${b['Occupied SKU']||b.skusInBin||'—'}</td>
      <td style="font-size:11px;color:var(--slate5)">${b['Last Updated']||'—'}</td>
      <td><select class="ti" style="width:100px;font-size:11px" onchange="updateBinStatusLocal('${id}',this.value)">
        ${['EMPTY','OCCUPIED','RESERVED'].map(s=>`<option ${st===s?'selected':''}>${s}</option>`).join('')}
      </select></td>
    </tr>`;
  }).join('');
}

function brandZoneLabelJS(b){
  const k=(b||'').toUpperCase().replace(/[\s_-]/g,'');
  if(k.includes('BEBODYWISE')||k.includes('BW')||k.includes('BB')) return 'R4–R13';
  if(k.includes('LITTLEJOYS')||k.includes('LJ')) return 'R14–R20';
  if(k.includes('MANMATTERS')||k.includes('MM')) return 'R21–R25';
  return 'R1–R3';
}

function initCC(){

  $('cc-bins-dl').innerHTML=bins.map(b=>`<option value="${b}">`).join('');

function gpBlock(sku,batch){return gatepasses.filter(r=>r.skuCode===sku).reduce((s,r)=>s+r.qty,0);}

function ccByBin(){
  const bin=$('cc-bin').value.trim(); if(!bin)return;
  const rows=inventory.filter(r=>(r.bin||r['Bin / Shelf']||'')===bin);
  if(!rows.length){
    $('cc-search-msg').innerText='No inventory on this shelf.';
    hideCCLines(); showAddStock(bin, false);
    renderCCOthers();
    return;
  }
  const skus=[...new Set(rows.map(r=>r.skuCode||r['SKU Code']||''))];
  if(skus.length===1) $('cc-sku').value=skus[0];
  $('cc-search-msg').innerText='';
  $('cc-add-card').style.display='none';
  loadCCLines(rows, bin);
}
function ccBySKU(){
  const q=$('cc-sku').value.trim(); if(!q)return;
  const rows=inventory.filter(r=>{const s=r.skuCode||r['SKU Code']||'';return s===q||s.toLowerCase().includes(q.toLowerCase())||String(r.skuName||r['SKU Name']||'').toLowerCase().includes(q.toLowerCase());});
  if(!rows.length){$('cc-search-msg').innerText='No inventory for: '+q;hideCCLines();return;}
  const bins=[...new Set(rows.map(r=>r.bin||r['Bin / Shelf']||''))];
  if(bins.length===1) $('cc-bin').value=bins[0];
  $('cc-search-msg').innerText='';
  loadCCLines(rows, $('cc-bin').value.trim()||null);
}

function loadCCLines(rows, binFilter){
  const filtered=binFilter?rows.filter(r=>(r.bin||r['Bin / Shelf']||'')===binFilter):rows;
  ccLines=filtered.map(r=>({
    bin:r.bin||r['Bin / Shelf']||'',
    skuCode:r.skuCode||r['SKU Code']||'',
    skuName:r.skuName||r['SKU Name']||'',
    ean:r.ean||r['EAN']||'',
    batch:r.batch||r['Batch No']||'',
    mfgDate:r.mfgDate||r['MFG Date']||'',
    expDate:r.expDate||r['EXP Date']||'',
    stockType:r.stockType||r['Stock Type']||'GOOD',
    systemQty:Number(r.qty||r['Qty']||0),
    gpBlock:gpBlock(r.skuCode||r['SKU Code']||'', r.batch||''),
    countedQty:Number(r.qty||r['Qty']||0),
    remarks:''
  }));
  $('cc-lines-card').style.display='block';
  $('cc-bin-label').innerText=binFilter||'All';
  // Show add stock card below (partial shelf — already has stock)
  if(binFilter) showAddStock(binFilter, true);
  renderCCLines(); renderCCOthers();
}

async function saveNewStock(){
  const bin=$('cc-bin').value.trim();
  const skuCode=$('ns-sku').value.trim(); const skuName=$('ns-name').value.trim();
  const batch=$('ns-batch').value.trim(); const mfgDate=$('ns-mfg').value.trim();
  const expDate=$('ns-exp').value.trim(); const qty=Number($('ns-qty').value)||0;
  const stockType=$('ns-type').value; const ean=$('ns-ean').value.trim();
  const msg=$('ns-msg');
  if(!bin){msg.innerText='Search a shelf first';return;}
  if(!skuCode){msg.innerText='SKU Code required';return;}
  if(!batch){msg.innerText='Batch No required';return;}
  if(qty<=0){msg.innerText='Qty must be > 0';return;}
  msg.innerText='';
  // Update inventory local
  const ex=inventory.find(r=>(r.bin||r['Bin / Shelf']||'')===bin&&(r.skuCode||r['SKU Code']||'')===skuCode&&(r.batch||r['Batch No']||'')===batch&&(r.stockType||r['Stock Type']||'GOOD')===stockType);
  if(ex){ex.qty=(Number(ex.qty||0)+qty);}
  else{ inventory.push({bin,skuCode,skuName,batch,mfgDate,expDate,stockType,qty,ean}); }
  lss('inv',inventory); syncBinsLocal();
  // Adjustment log
  const now=new Date().toISOString();
  adjLog.push({time:now,bin,skuCode,skuName,batch,mfgDate,expDate,stockType,qty:+qty,action:'ADD',ref:'New stock via Cycle Count',by:U.name});
  lss('adj',adjLog); renderAdjLog(); updateAdjStats();
  showResult('cc-result','✅ Added '+qty+' units of '+skuCode+' ('+batch+') to '+bin,'ok');
  $('ns-qty').value=''; ccByBin();
  await call('uploadInventoryDump',inventory);
  await call('saveAdjustment',adjLog);
}

async function doTransfer(fromBin,toBin,sku,batch,stockType,maxQty,sid,mfgDate,expDate){
  if(!toBin){alert('Search a destination shelf first');return;}
  const qEl=$('tq_'+sid); const moveQty=qEl?Number(qEl.value)||0:maxQty;
  if(moveQty<=0){alert('Enter qty to move');return;}
  if(moveQty>maxQty){alert('Cannot move more than available: '+maxQty);return;}
  if(fromBin===toBin){alert('Same shelf');return;}
  if(!confirm(`Move ${moveQty} units of ${sku} (${batch})\nFrom: ${fromBin}\nTo:   ${toBin}`))return;
  const now=new Date().toISOString();
  // Update inventory
  let rem=moveQty;
  inventory.forEach(r=>{if((r.bin||r['Bin / Shelf']||'')===fromBin&&(r.skuCode||r['SKU Code']||'')===sku&&(r.batch||r['Batch No']||'')===batch&&rem>0){const cut=Math.min(rem,Number(r.qty||0));r.qty=Number(r.qty||0)-cut;rem-=cut;}});
  inventory=inventory.filter(r=>Number(r.qty||0)>0);
  const ex=inventory.find(r=>(r.bin||r['Bin / Shelf']||'')===toBin&&(r.skuCode||r['SKU Code']||'')===sku&&(r.batch||r['Batch No']||'')===batch&&(r.stockType||r['Stock Type']||'GOOD')===stockType);
  if(ex){ex.qty=Number(ex.qty||0)+moveQty;}
  else{const src=inventory.find(r=>(r.skuCode||r['SKU Code']||'')===sku&&(r.batch||r['Batch No']||'')===batch)||{};inventory.push({bin:toBin,skuCode:sku,skuName:src.skuName||src['SKU Name']||sku,batch,stockType,qty:moveQty,mfgDate,expDate,ean:src.ean||''});}
  lss('inv',inventory); syncBinsLocal();
  // Adj log
  const skuName=inventory.find(r=>(r.skuCode||r['SKU Code']||'')===sku)?.skuName||sku;
  adjLog.push({time:now,bin:fromBin,skuCode:sku,skuName,batch,mfgDate,expDate,stockType,qty:-moveQty,action:'REMOVE',ref:'Moved to '+toBin,by:U.name});
  adjLog.push({time:now,bin:toBin,  skuCode:sku,skuName,batch,mfgDate,expDate,stockType,qty:+moveQty,action:'ADD',   ref:'Moved from '+fromBin,by:U.name});
  lss('adj',adjLog); renderAdjLog(); updateAdjStats();
  showResult('cc-result','✅ Moved '+moveQty+' units of '+sku+' from '+fromBin+' → '+toBin,'ok');
  ccByBin();
  await call('uploadInventoryDump',inventory);
  await call('saveAdjustment',adjLog);
}

async function saveCountLines(){
  if(!ccLines.length){alert('No count lines to save');return;}
  const now=new Date().toISOString();
  const newEntries=[];
  ccLines.forEach(l=>{
    const diff=l.countedQty-l.systemQty;
    const entry={time:now,bin:l.bin,skuCode:l.skuCode,skuName:l.skuName||'',batch:l.batch,systemQty:l.systemQty,countedQty:l.countedQty,diff,mfgDate:l.mfgDate,expDate:l.expDate,stockType:l.stockType||'GOOD',gpBlock:l.gpBlock||0,remarks:l.remarks||'',by:U.name};
    ccLog.push(entry); newEntries.push(entry);
  });
  updateCCStats(); renderCCLog();
  const diffs=ccLines.filter(l=>l.countedQty!==l.systemQty).length;
  showResult('cc-result','Count saved. '+diffs+' difference(s) recorded.',diffs?'warn':'ok');
  const r=await call('saveCycleCount',newEntries);
  if(r&&r.success) showResult('cc-result','Count saved to Google Sheet. '+diffs+' difference(s).',diffs?'warn':'ok');
}
